function headsOrTails(){
  if(Math.round(Math.random() * 100) > 30){
    return 1
  }
  return 0
}

let heads = 0;
let tails = 0;

for(let i = 0; i < 1000; i++){
  if(headsOrTails()){
    heads++
  }
  else{
    tails++
  }
}

console.log(`Орёл выпал ${heads} раз, решка- ${tails} раз`)

function decimalPart(number){
    number = String(number)
    decimalpart = ''
    for(let i = number.indexOf('.') + 1; i < number.length; i++){
        decimalpart += number[i]
    }
    return decimalpart
}
